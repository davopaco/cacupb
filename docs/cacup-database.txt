CREATE TABLE IF NOT EXISTS `USUARIOS` (
  `ID` INT NOT NULL,
  `HASH` VARCHAR(64) NOT NULL,
  `NOMBRES` VARCHAR(100) NOT NULL,
  `APELLIDOS` VARCHAR(100) NOT NULL,
  `TIPO` TINYINT NOT NULL,
  PRIMARY KEY (`ID`));

CREATE TABLE IF NOT EXISTS `CLIENTES` (
  `ID` INT NOT NULL,
  `NOMBRES` VARCHAR(100) NOT NULL,
  `APELLIDOS` VARCHAR(100) NOT NULL,
  `FECHA_NAC` DATE NOT NULL,
  `DIRECCION` VARCHAR(250) NOT NULL,
  PRIMARY KEY (`ID`));

CREATE TABLE IF NOT EXISTS `SEDES` (
  `ID` INT NOT NULL AUTO_INCREMENT,
  `NOMBRE` VARCHAR(200) NOT NULL,
  PRIMARY KEY (`ID`));

CREATE TABLE IF NOT EXISTS `CITAS` (
  `ID` INT NOT NULL AUTO_INCREMENT,
  `FECHA` TIMESTAMP NOT NULL,
  `DESCRIPCION` TEXT NOT NULL,
  `TIPO` ENUM("REC", "DEV", "ASE") NOT NULL,
  `HORA` TIME NOT NULL,
  `CLIENTES_ID` INT NOT NULL,
  `ESTADO` TINYINT NOT NULL COMMENT 'Se refiere al estado de la cita en el momento. En este caso 0 es \"por asistir\", 1 es \"asistida\", 2 es \"cancelada\" y 3 es \"no asistida\".',
  `SEDES_ID` INT NOT NULL,
  PRIMARY KEY (`ID`),
  INDEX `fk_CITAS_CLIENTES_idx` (`CLIENTES_ID` ASC) VISIBLE,
  INDEX `fk_CITAS_SEDES1_idx` (`SEDES_ID` ASC) VISIBLE,
  CONSTRAINT `fk_CITAS_CLIENTES`
    FOREIGN KEY (`CLIENTES_ID`)
    REFERENCES `CLIENTES` (`ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_CITAS_SEDES1`
    FOREIGN KEY (`SEDES_ID`)
    REFERENCES `SEDES` (`ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
